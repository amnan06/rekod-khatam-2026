function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Murid'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).map(row => { let obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const payload = JSON.parse(e.postData.contents); const action = payload.action; if (action === 'tambah') { const sheet = ss.getSheetByName('Murid'); sheet.appendRow([ payload.id, payload.nama, payload.jantina, payload.status, payload.ms, payload.daftar ]); } if (action === 'kemaskini') { const sheet = ss.getSheetByName('Murid'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == payload.id) { sheet.getRange(i+1, 1, 1, 6).setValues([[ payload.id, payload.nama, payload.jantina, payload.status, payload.ms, payload.daftar ]]); break; } } } if (action === 'rekod') { const sheet = ss.getSheetByName('Rekod'); sheet.appendRow([ payload.muridId, payload.minggu, payload.jenis, payload.ms, payload.catatan, payload.tarikh ]); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
